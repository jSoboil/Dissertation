model {
# Sub-model 3:
 for (i in 1:Nstud) {
  rA[i] ~ dbin(pA[i], nA[i])
  rB[i] ~ dbin(pB[i], nB[i])
 
  logit(pA[i]) <- mu[i]
  logit(pB[i]) <- mu[i] + delta[i]
 
  mu[i] ~ dnorm(0, 1.0e-5)
  delta[i] ~ dnorm(d, prec)
}
# Priors for sub-model 3:
 d ~ dnorm(0, 1.0e-6)
 tau ~ dunif(0, 10)
 tau.sq <- (tau * tau)
 prec <- 1 / tau.sq
 
 OR <- exp(d)
 pEfficacy <- (OR / 1 + OR)
 
}
