
 model {
  for (i in 1:n_t) {
   # Likelihood:
   r_Age[i] ~ dbin(omega_Age[i], n_Age[i])
   # Logit function:
   logit(omega_Age[i]) <- nu[i] + eta[i]
  
  # Random effect:
  eta[i] ~ dnorm(0, 1.0e-5)
  # Informative prior:
  nu[i] ~ dnorm(iota, phi.prec)
  
  }
  # Hyperpriors on precision:
  iota ~ dnorm(0, 1.0e-6)
  phi ~ dunif(0, 10)
  phi.prec <- 1 / (phi * phi)
  
}
