
 # Sub-model 1: Population level incidence for Ages 16-23:
   model {
    for (i in 1:length(r)) {
     # Binomial Likelihood on 
     # postive outcome:
     r[i] ~ dbin(p[i], n[i])
  
     # Link function between
     # Age and Outcome:
     p[i] <- p_Age[i]
   
     # Normal Likelihood on 
     # Age:
     p_Age[i] ~ dnorm(mu, prec)
   }
  
 # Priors on mu and sd:
 mu ~ dnorm(0, .1)
 sd ~ dunif(0, 10)
 tau.sq <- sd * sd
 prec <- 1 / tau.sq
  
 # Priors on LSIL and
 # HSIL
 LSIL ~ dbeta(.2, .8)
 HSIL ~ dbeta(.35, .9)
 
}
